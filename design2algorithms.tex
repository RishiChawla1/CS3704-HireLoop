\begin{algorithm}
\caption{The \textbf{matchApplicantsToJobs()} function}
\textbf{Input:} The list of applicants, and list of jobs as well as their criteria

\textbf{Output:} A one-to-one map between best jobs and applicants

Create applicant and job opening nodes;


\ForEach{user and job opening}{
assign them to one bipartite graph node.

use \textbf{matchCriteriaToScore()} to determine weights between jobs and applicants.

assign that weight to the node.
}
	
\While{There are still job openings}{
	assign matches between one applicant and one job opening based on nearest fit job;

	remove them from the graph and notify applicant and employer of match;
}
\end{algorithm}
